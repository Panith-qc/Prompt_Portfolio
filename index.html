class PE_DD_Chain:
    def execute_full_dd(self, deal_params):
        results = {}
        
        # Phase 1: Initial Screening (15 minutes)
        for prompt in self.prompts['screening']:
            results[prompt] = self.execute_prompt(prompt, deal_params)
            if results[prompt]['recommendation'] == 'PASS':
                return {'decision': 'PASS', 'reason': results[prompt]['rationale']}
        
        # Phase 2: Deep Dive Analysis (120 minutes)
        analysis_phases = ['commercial', 'financial', 'management', 'risk']
        for phase in analysis_phases:
            results[phase] = self.analyze_phase(phase, deal_params)
        
        return self.generate_investment_memo(results)

import json

from typing import Dict, List, Optional

# Prompt Template Structure
@dataclass
class PromptTemplate:
    id: str
    name: str
    category: str
    version: str
    template: str
    variables: List[str]
    performance_metrics: Dict

# Variable Templating System
class VariableTemplater:
    def render_prompt(self, template: str, variables: Dict[str, str]) -> str:
        rendered = template
        for key, value in variables.items():
            placeholder = "{" + key + "}"
            rendered = rendered.replace(placeholder, value)
        return rendered

📈 A/B Testing Framework

class PerformanceTester:
    def run_ab_test(self, prompt_a: str, prompt_b: str, test_cases: List[Dict]):
        results = {
            "prompt_a": {"scores": [], "times": []},
            "prompt_b": {"scores": [], "times": []}
        }
        
        for test_case in test_cases:
            # Test both prompts
            score_a, time_a = self.evaluate_prompt(prompt_a, test_case)
            score_b, time_b = self.evaluate_prompt(prompt_b, test_case)
            
            results["prompt_a"]["scores"].append(score_a)
            results["prompt_a"]["times"].append(time_a)
            results["prompt_b"]["scores"].append(score_b)
            results["prompt_b"]["times"].append(time_b)
        
        return self.calculate_statistical_significance(results)
